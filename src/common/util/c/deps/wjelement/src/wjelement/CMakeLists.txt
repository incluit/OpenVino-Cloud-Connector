	element.c
	element.h
	search.c
	types.c
	schema.c
	hash.c
)

	wjreader
	wjwriter
	xpl
	${ALL_LIBS}
)

	wjelement
	PROPERTIES
	VERSION 1.0.0
	SOVERSION 1
	INSTALL_NAME_DIR "${LIB_DEST_DIR}"
)

	LIBRARY DESTINATION ${LIB_DEST_DIR}
	ARCHIVE DESTINATION ${LIB_DEST_DIR}
	)
